# Calculate RUL: for each engine, RUL = max_cycle - current_cycle
def calculate_rul(group):
    group['RUL'] = group[1].max() - group[1]
    return group

data_train_FD1 = data_train_FD1.groupby(0, group_keys=False).apply(calculate_rul)
